      import React, { useState } from 'react';
      import { ShoppingCart, Heart, Search, Sparkles, Trash2, X, CheckCircle } from 'lucide-react';
      const PRODUCTS = [
      { id: 1, name: "Áo thun KienStore Sáng Tạo v1", price: 290000, image: "https://images.unsplash.com/photo-1521572267360-ee0c2909d518?w=500&auto=format&fit=crop&q=60", category: "Thời trang" },
      { id: 2, name: "Sổ tay Concept Thiết Kế", price: 85000, image: "https://images.unsplash.com/photo-1531346878377-a5be20888e57?w=500&auto=format&fit=crop&q=60", category: "Văn phòng phẩm" },
      { id: 3, name: "Bình nước giữ nhiệt KienStore", price: 350000, image: "https://images.unsplash.com/photo-1602143407151-7111542de6e8?w=500&auto=format&fit=crop&q=60", category: "Phụ kiện" },
      { id: 4, name: "Balo Canvas Mini Đa Năng", price: 420000, image: "https://images.unsplash.com/photo-1548036328-c9fa89d128fa?w=500&auto=format&fit=crop&q=60", category: "Phụ kiện" },
      { id: 5, name: "Ốp lưng điện thoại Artist Series", price: 150000, image: "https://images.unsplash.com/photo-1603302576837-37561b2e2302?w=500&auto=format&fit=crop&q=60", category: "Phụ kiện" },
      { id: 6, name: "Set Sticker KienStore Đặc Biệt", price: 45000, image: "https://images.unsplash.com/photo-1572375995501-4b0894dbe050?w=500&auto=format&fit=crop&q=60", category: "Quà tặng" }
      ];
      export default function App() {
      const [cart, setCart] = useState([]);
      const [isCartOpen, setIsCartOpen] = useState(false);
      const [searchTerm, setSearchTerm] = useState('');
      const [selectedCategory, setSelectedCategory] = useState('Tất cả');
      const [creativeText, setCreativeText] = useState('');
      const [customizedProducts, setCustomizedProducts] = useState([]);
      const addToCart = (product) => {
      setCart((prevCart) => {
      const existingItem = prevCart.find(item => item.id === product.id);
      if (existingItem) {
      return prevCart.map(item =>
      item.id === product.id ? { ...item, quantity: item.quantity + 1 } : item
      );
      }
      return [...prevCart, { ...product, quantity: 1 }];
      });
      };
      const removeFromCart = (productId) => {
      setCart(prevCart => prevCart.filter(item => item.id !== productId));
      };
      const updateQuantity = (productId, amount) => {
      setCart(prevCart =>
      prevCart.map(item => {
      if (item.id === productId) {
      const newQty = item.quantity + amount;
      return newQty > 0 ? { ...item, quantity: newQty } : item;
      }
      return item;
      })
      );
      };
      const handleCreateCustomProduct = (e) => {
      e.preventDefault();
      if (!creativeText.trim()) return;
      const newProduct = {
      id: Date.now(),
      name: `Áo Custom: "${creativeText}"`,
      price: 320000,
      image: "https://images.unsplash.com/photo-1521572267360-ee0c2909d518?w=500&auto=format&fit=crop&q=60",
      category: "Sáng tạo cá nhân"
      };
      setCustomizedProducts([newProduct, ...customizedProducts]);
      setCreativeText('');
      };
      const filteredProducts = [...customizedProducts, ...PRODUCTS].filter(product => {
      const matchesSearch = product.name.toLowerCase().includes(searchTerm.toLowerCase());
      const matchesCategory = selectedCategory === 'Tất cả' || product.category === selectedCategory;
      return matchesSearch && matchesCategory;
      });
      const cartTotal = cart.reduce((sum, item) => sum + (item.price * item.quantity), 0);
      const cartItemCount = cart.reduce((sum, item) => sum + item.quantity, 0);
      return (
      <div className="min-h-screen bg-slate-50 text-slate-800 font-sans selection:bg-indigo-500 selection:text-white">
      <nav className="bg-white/80 backdrop-blur-md border-b border-slate-100 sticky top-0 z-40 transition-all">
      <div className="max-w-7xl mx-auto px-4 sm:px-6 lg:px-8 h-16 flex items-center justify-between">
      <div className="flex items-center gap-2">
      <span className="text-2xl font-black tracking-wider text-indigo-600">KIENSTORE</span>
      <span className="bg-indigo-50 text-indigo-700 text-xs font-bold px-2.5 py-1 rounded-full flex items-center gap-1">
      <Sparkles className="w-3.5 h-3.5" /> Creative
      </span>
      </div>
      <div className="hidden md:flex items-center gap-8">
      <a href="#shop" className="hover:text-indigo-600 font-medium transition-colors">Cửa hàng</a>
      <a href="#creative" className="hover:text-indigo-600 font-medium transition-colors">Khu Sáng Tạo</a>
      </div>
      <div className="flex items-center gap-4">
      <button
      onClick={() => setIsCartOpen(true)}
      className="relative p-2.5 hover:bg-slate-100 rounded-full transition-colors group"
      >
      <ShoppingCart className="w-6 h-6 text-slate-700 group-hover:text-indigo-600 transition-colors" />
      {cartItemCount > 0 && (
      <span className="absolute -top-1 -right-1 bg-rose-500 text-white text-xs w-5 h-5 flex items-center justify-center rounded-full font-bold animate-bounce">
      {cartItemCount}
      </span>
      )}
      </button>
      </div>
      </div>
      </nav>
      <header className="relative bg-gradient-to-br from-indigo-900 via-indigo-850 to-purple-900 text-white py-24 px-4 overflow-hidden">
      <div className="absolute inset-0 bg-[radial-gradient(circle_at_top_right,rgba(120,119,198,0.3),transparent_50%)]"></div>
      <div className="max-w-4xl mx-auto text-center relative z-10">
      <h1 className="text-5xl md:text-7xl font-black mb-6 tracking-tight leading-none bg-clip-text text-transparent bg-gradient-to-r from-white via-slate-100 to-indigo-200">
      KIENSTORE
      </h1>
      <p className="text-lg md:text-xl text-indigo-100/90 max-w-2xl mx-auto mb-10 font-normal leading-relaxed">
      Nơi hội tụ những ý tưởng thiết kế độc bản, kết hợp mua sắm và tự do cá nhân hóa phong cách của riêng bạn.
      </p>
      <div className="flex flex-wrap gap-4 justify-center">
      <a href="#shop" className="bg-white text-indigo-900 font-bold px-8 py-3.5 rounded-full hover:shadow-xl hover:scale-105 active:scale-95 transition duration-200">
      Mua sắm ngay
      </a>
      <a href="#creative" className="bg-indigo-600/30 border border-indigo-400/30 text-white font-bold px-8 py-3.5 rounded-full hover:bg-indigo-600/50 transition duration-200">
      Tự thiết kế áo
      </a>
      </div>
      </div>
      </header>
      <section id="creative" className="max-w-7xl mx-auto px-4 sm:px-6 lg:px-8 py-16">
      <div className="bg-white rounded-3xl p-8 md:p-12 shadow-sm border border-slate-100">
      <div className="max-w-2xl mx-auto text-center">
      <span className="text-indigo-600 font-bold text-sm tracking-widest uppercase">Không gian cá nhân</span>
      <h2 className="text-3xl font-black mt-2 mb-4">Tự Thiết Kế Sản Phẩm Cho Riêng Bạn</h2>
      <p className="text-slate-500 mb-8">Nhập thông điệp hoặc ý tưởng bạn muốn in lên áo thun độc quyền KienStore.</p>
      <form onSubmit={handleCreateCustomProduct} className="flex flex-col sm:flex-row gap-3">
      <input
      type="text"
      value={creativeText}
      onChange={(e) => setCreativeText(e.target.value)}
      placeholder="Ví dụ: 'Sáng tạo hay là chết', 'Kien 1998'..."
      className="flex-1 px-6 py-4 rounded-2xl border border-slate-200 focus:outline-none focus:ring-2 focus:ring-indigo-500 text-lg transition-all"
      />
      <button
      type="submit"
      className="bg-indigo-600 hover:bg-indigo-700 text-white font-bold px-8 py-4 rounded-2xl transition duration-200 shadow-lg shadow-indigo-600/20 active:scale-95"
      >
      Tạo sản phẩm
      </button>
      </form>
      </div>
      </div>
      </section>
      <main id="shop" className="max-w-7xl mx-auto px-4 sm:px-6 lg:px-8 py-12 border-t border-slate-100">
      <div className="flex flex-col md:flex-row md:items-center justify-between gap-6 mb-12">
      <div>
      <h2 className="text-3xl font-black tracking-tight">Danh mục sản phẩm</h2>
      <p className="text-slate-500 mt-1">Khám phá các sản phẩm độc quyền chất lượng cao</p>
      </div>
      <div className="flex flex-wrap gap-3 items-center">
      <div className="relative w-full sm:w-64">
      <Search className="w-5 h-5 text-slate-400 absolute left-4 top-1/2 -translate-y-1/2" />
      <input
      type="text"
      placeholder="Tìm kiếm sản phẩm..."
      value={searchTerm}
      onChange={(e) => setSearchTerm(e.target.value)}
      className="w-full pl-11 pr-4 py-2.5 rounded-2xl border border-slate-200 focus:outline-none focus:ring-2 focus:ring-indigo-500 transition-all text-sm"
      />
      </div>
      <div className="flex gap-2 bg-slate-100 p-1 rounded-2xl overflow-x-auto">
      {['Tất cả', 'Thời trang', 'Phụ kiện', 'Văn phòng phẩm', 'Sáng tạo cá nhân'].map((category) => (
      <button
      key={category}
      onClick={() => setSelectedCategory(category)}
      className={`px-4 py-1.5 rounded-xl text-xs font-bold transition-all whitespace-nowrap ${
      selectedCategory === category
      ? 'bg-white text-indigo-600 shadow-sm'
      : 'text-slate-500 hover:text-slate-800'
      }`}
      >
      {category}
      </button>
      ))}
      </div>
      </div>
      </div>
      {filteredProducts.length === 0 ? (
      <div className="text-center py-20 bg-white rounded-3xl border border-slate-100">
      <p className="text-slate-400 text-lg">Không tìm thấy sản phẩm nào khớp với bộ lọc.</p>
      </div>
      ) : (
      <div className="grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-3 gap-8">
      {filteredProducts.map((product) => (
      <div key={product.id} className="bg-white rounded-3xl border border-slate-100 overflow-hidden group hover:shadow-xl hover:shadow-slate-150/40 transition duration-300 flex flex-col h-full">
      <div className="relative aspect-square overflow-hidden bg-slate-100">
      <img src={product.image} alt={product.name} className="w-full h-full object-cover group-hover:scale-105 transition duration-500" />
      {product.category === "Sáng tạo cá nhân" && (
      <span className="absolute top-4 left-4 bg-purple-600 text-white text-xs font-bold px-3 py-1.5 rounded-full shadow-lg flex items-center gap-1">
      <Sparkles className="w-3 h-3" /> Thiết kế của bạn
      </span>
      )}
      </div>
      <div className="p-6 flex flex-col flex-1 justify-between">
      <div>
      <span className="text-xs font-bold text-indigo-600 tracking-wider uppercase">{product.category}</span>
      <h3 className="font-bold text-lg mt-1 text-slate-800 group-hover:text-indigo-600 transition-colors line-clamp-2">{product.name}</h3>
      </div>
      <div className="flex items-center justify-between mt-6 pt-4 border-t border-slate-50">
      <span className="text-xl font-black text-slate-900">{product.price.toLocaleString('vi-VN')} đ</span>
      <button
      onClick={() => addToCart(product)}
      className="bg-slate-900 hover:bg-indigo-600 text-white px-5 py-2.5 rounded-2xl text-sm font-bold transition-all duration-200 active:scale-95"
      >
      Thêm vào giỏ
      </button>
      </div>
      </div>
      </div>
      ))}
      </div>
      )}
      </main>
      {isCartOpen && (
      <div className="fixed inset-0 z-50 overflow-hidden">
      <div className="absolute inset-0 bg-slate-900/40 backdrop-blur-xs transition-opacity" onClick={() => setIsCartOpen(false)}></div>
      <div className="absolute inset-y-0 right-0 max-w-full flex pl-10">
      <div className="w-screen max-w-md bg-white shadow-2xl flex flex-col h-full">
      <div className="px-6 py-5 border-b border-slate-100 flex items-center justify-between">
      <h2 className="text-lg font-bold text-slate-900 flex items-center gap-2">
      <ShoppingCart className="w-5 h-5" /> Giỏ hàng của bạn ({cartItemCount})
      </h2>
      <button onClick={() => setIsCartOpen(false)} className="p-1 hover:bg-slate-100 rounded-full transition-colors">
      <X className="w-6 h-6 text-slate-400 hover:text-slate-600" />
      </button>
      </div>
      <div className="flex-1 py-6 overflow-y-auto px-6">
      {cart.length === 0 ? (
      <div className="h-full flex flex-col items-center justify-center text-center">
      <ShoppingCart className="w-16 h-16 text-slate-300 mb-4" />
      <p className="text-slate-500 font-medium">Giỏ hàng của bạn đang trống</p>
      <button
      onClick={() => setIsCartOpen(false)}
      className="mt-4 text-sm font-bold text-indigo-600 hover:text-indigo-700"
      >
      Tiếp tục mua sắm &rarr;
      </button>
      </div>
      ) : (
      <div className="space-y-6">
      {cart.map((item) => (
      <div key={item.id} className="flex py-2 border-b border-slate-50 pb-4">
      <div className="h-20 w-20 flex-shrink-0 overflow-hidden rounded-2xl border border-slate-100">
      <img src={item.image} alt={item.name} className="h-full w-full object-cover" />
      </div>
      <div className="ml-4 flex flex-1 flex-col">
      <div>
      <div className="flex justify-between text-base font-semibold text-slate-900">
      <h3 className="line-clamp-1">{item.name}</h3>
      <p className="ml-4 whitespace-nowrap">{(item.price * item.quantity).toLocaleString('vi-VN')} đ</p>
      </div>
      <p className="mt-1 text-xs text-slate-400">{item.category}</p>
      </div>
      <div className="flex flex-1 items-end justify-between text-sm">
      <div className="flex items-center border border-slate-200 rounded-xl overflow-hidden bg-slate-50">
      <button onClick={() => updateQuantity(item.id, -1)} className="px-3 py-1 hover:bg-slate-200 text-slate-600 font-bold transition-colors">-</button>
      <span className="px-3 text-slate-800 font-bold text-xs">{item.quantity}</span>
      <button onClick={() => updateQuantity(item.id, 1)} className="px-3 py-1 hover:bg-slate-200 text-slate-600 font-bold transition-colors">+</button>
      </div>
      <button onClick={() => removeFromCart(item.id)} className="font-bold text-rose-500 hover:text-rose-600 flex items-center gap-1 transition-colors">
      <Trash2 className="w-4 h-4" /> Xóa
      </button>
      </div>
      </div>
      </div>
      ))}
      </div>
      )}
      </div>
      {cart.length > 0 && (
      <div className="border-t border-slate-100 px-6 py-6 bg-slate-50">
      <div className="flex justify-between text-base font-bold text-slate-900 mb-2">
      <p>Tổng tiền tạm tính</p>
      <p className="text-xl text-indigo-600">{cartTotal.toLocaleString('vi-VN')} đ</p>
      </div>
      <p className="text-xs text-slate-400 mb-6">Phí vận chuyển và thuế sẽ được tính khi thanh toán.</p>
      <button
      onClick={() => alert('Chức năng thanh toán sẽ được kết nối với API Sandbox tại bước tiếp theo!')}
      className="w-full flex items-center justify-center rounded-2xl border border-transparent bg-indigo-600 px-6 py-4 text-base font-bold text-white shadow-lg shadow-indigo-600/20 hover:bg-indigo-700 transition duration-200"
      >
      Thanh toán đơn hàng
      </button>
      </div>
      )}
      </div>
      </div>
      </div>
      )}
      <footer className="bg-slate-900 text-slate-400 py-16 border-t border-slate-800">
      <div className="max-w-7xl mx-auto px-4 sm:px-6 lg:px-8 text-center md:text-left">
      <div className="grid grid-cols-1 md:grid-cols-3 gap-12">
      <div>
      <span className="text-2xl font-black text-white tracking-wider">KIENSTORE</span>
      <p className="mt-4 text-sm max-w-xs text-slate-400 leading-relaxed">
      Nền tảng mua sắm kết hợp thiết kế tự chọn, kiến tạo giá trị phong cách độc đáo cho cộng đồng trẻ sáng tạo.
      </p>
      </div>
      <div>
      <h3 className="text-sm font-semibold text-slate-200 tracking-wider uppercase mb-4">Danh Mục</h3>
      <ul className="space-y-2.5 text-sm">
      <li><a href="#shop" className="hover:text-white transition-colors">Tất cả sản phẩm</a></li>
      <li><a href="#creative" className="hover:text-white transition-colors">Không gian tự custom</a></li>
      </ul>
      </div>
      <div>
